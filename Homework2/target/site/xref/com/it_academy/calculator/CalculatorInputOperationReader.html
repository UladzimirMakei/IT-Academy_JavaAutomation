9       public static char inputOperation() {
10          boolean successfulInput = false;
11          String operationInput;
12          do {
13              operationInput = greetingAndScanning ();
14              if(correctInput(operationInput)) {
15                  successfulInput = true;
16              } else {
17                  out.println("Your input is not correct. Please try again");
18              }
19          } while (!successfulInput);
20          return operationInput.charAt(0);
21      }
22  
23      public static String greetingAndScanning () {
24          out.println("Enter arithmetic operation '+' '-' '*' '/': ");
25          Scanner scanner = new Scanner(System.in);
26          return scanner.nextLine();
27      }
28  
29      public static boolean correctInput (String operationInput) {
30          return (operationInput.length()==1)&&(operationInput.charAt(0) == '+' || operationInput.charAt(0) == '-'
31                  || operationInput.charAt(0) == '/' || operationInput.charAt(0) == '*');
32      }
