9       public static double inputValue() throws NumberFormatException {
10          boolean successfulInput = false;
11          String valueInput ="0";
12          do {
13              try {
14                  valueInput = greetingAndScanning();
15                  successfulInput = correctInput(valueInput);
16              } catch (NumberFormatException e) {
17                  out.println("Your input is not correct. Please try again");
18              }
19          } while (!successfulInput);
20          return Double.parseDouble(valueInput);
21      }
22  
23      public static String greetingAndScanning() {
24          out.println("Enter a number: ");
25          Scanner scanner = new Scanner(System.in);
26          return scanner.nextLine().replace(',', '.');
27      }
28  
29      public static boolean correctInput(String value) {
30          boolean result;
31          if (!Double.isNaN(Double.parseDouble(value)) && !Double.isInfinite(Double.parseDouble(value))){
32              result = true;
33          } else {
34              out.println("Your input is not correct. Please try again");
35              result = false;
36          }
37          return result;
38      }
